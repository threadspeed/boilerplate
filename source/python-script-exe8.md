```python
nome = []
idades = []
alturas = []
for nomes in range(2):
    nomes = input('Digite o nome: ')
    nome.append(nomes)
    idade = int(input('Digite sua idade %s\n'%nomes))
    idades.append(idade)
    altura = float(input('Digite sua altura\n'))
    alturas.append(altura)
idades.reverse()
alturas.reverse()
print(idades)
print(alturas)

```
