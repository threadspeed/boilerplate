def prga0(s, n):
    """weakened prga"""
    i = j = 0
    res = bytearray()
    for _ in range(n):
        i = (i + 1) & mask
        j = (j + s[i]) & mask
        # s[i], s[j] = s[j], s[i]
        res.append(s[(s[i]+s[j])&mask])
    return tuple(res)
