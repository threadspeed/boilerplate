import time

def time_check(func):
    def new_func(*args):
        start_time = time.time()
        print("함수를 시작합니다")
        result = func(*args)
        print("함수를 끝냅니다.")
        end_time = time.time()
        print("{} 초가 걸렸습니다.".format(end_time-start_time))
        return result
    return new_func


def fib(n):
    if n<= 2:
        return 1
    elif n==3:
        return 2
    else:
        return 2*fib(n-2)+fib(n-3)

@time_check
def find_fib(n):
    return fib(n)

print(find_fib(35))
